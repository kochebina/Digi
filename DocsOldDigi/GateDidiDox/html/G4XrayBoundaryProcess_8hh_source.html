132#ifdef GATE_USE_XRAYLIB
133G4double G4XrayBoundaryProcess::GetRindex(G4Material *Material, G4double Energy) {
134#else
135G4double G4XrayBoundaryProcess::GetRindex(G4Material *Material, G4double) {
136#endif
137 G4double delta = 0.0;
138 G4double Density = Material->GetDensity() / (g / cm3);
139
140#ifdef GATE_USE_XRAYLIB
141#if XRAYLIB_MAJOR > 3
142 xrl_error *error = NULL;
143 for (unsigned int i = 0; i < Material->GetElementVector()->size(); ++i) {
144 delta += (1 - Refractive_Index_Re(Material->GetElementVector()->at(i)->GetSymbol(), Energy/(keV), 1.0,&error)) * Material->GetFractionVector()[i];
145 if (error != NULL) {
146 G4cerr << "error message: " << error->message << "\n";
147 xrl_clear_error(&error);
148 }
149 }
150#else
151 for (unsigned int i = 0; i < Material->GetElementVector()->size(); ++i)
152 delta += (1 - Refractive_Index_Re(Material->GetElementVector()->at(i)->GetSymbol(), Energy/(keV), 1.0)) * Material->GetFractionVector()[i];
153#endif
154#else
155 G4Exception( "G4XrayBoundaryProcess::GetRindex", "GetRindex", FatalException, "Xraylib is not available\n");
156#endif
157
158 return 1 - delta * Density;
159}
