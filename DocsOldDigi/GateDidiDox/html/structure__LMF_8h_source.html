40/* 1.A This structure gives the ID's format */
41 struct LMF_ccs_scanEncodingID { /* rrrssssmmmcccccl */
42 u8 bitForRsectors; /* number of bits reserved for rings/sectors in ID */
43 u16 maximumRsectors; /* maximum number of sectors = 2 ** bitForRsectors */
44 u8 bitForModules; /* number of bits reserved for modules in ID */
45 u16 maximumModules; /* maximum number of modules = 2**bitForModules */
46 u8 bitForSubmodules; /* number of bits reserved for submodules in ID */
47 u16 maximumSubmodules; /* maximum number of submodules = 2 ** bitForSubmodules */
48 u8 bitForCrystals; /* number of bits reserved for crystals in ID */
49 u16 maximumCrystals; /* maximum number of crystals = 2** bitForCrystals */
50 u8 bitForLayers; /* number of bits reserved for layers in ID */
51 u16 maximumLayers; /* maximum number of layers = 2** bitForLayers */
52 };

55/* 1.B This structure gives the Scan Design */
56 struct LMF_ccs_scannerTopology {
57 u16 numberOfRings; /* number of rings */
58 u16 numberOfSectors; /* number of sectors */
59 u16 totalNumberOfRsectors; /* number of rings * number of sectors */
60
61 /*-=-=-=- FOR EACH RSECTOR : =-=-=-=-=-=*/
62 u16 axialNumberOfModules; /* number of modules axially */
63 u16 tangentialNumberOfModules; /* number of modules tangentially */
64 u16 totalNumberOfModules; /* number of modules axially * tangentially */
65
66 /*-=-=-=- FOR EACH MODULE : =-=-=-=-=-=*/
67 u16 axialNumberOfSubmodules; /* number of submodules axially */
68 u16 tangentialNumberOfSubmodules; /* number of submodules tangentially */
69 u16 totalNumberOfSubmodules; /* number of submodules = Axial * Tangential */
70
71 /*-=-=-=- FOR EACH SUBMODULE : =-=-=-=-=-=*/
72 u16 axialNumberOfCrystals; /* number of crystals axially */
73 u16 tangentialNumberOfCrystals; /* number of crystals tangentially */
74 u16 totalNumberOfCrystals; /* number of crystals = Axial * Tangential */
75
76 /*-=-=-=- FOR EACH CRYSTAL : =-=-=-=-=-=*/
77
78 u16 axialNumberOfLayers; /* number of layers axially (always 1) */
79 u16 radialNumberOfLayers; /* number of layers radially */
80 u16 totalNumberOfLayers; /* number of layers */

84/* 1.C This structure gives the kind(s) of records the .ccs file do contain */
85 struct LMF_ccs_scanContent {
86 u8 nRecord; /* number of different records */
87 /* first record : event record */
88 u8 eventRecordBool; /* event recorded if 1 */
89 u8 eventRecordTag; /* event tag=0(1st bit of encoding event) */
90 /* second record : countrate record */
91 u8 countRateRecordBool; /* countrate recorded if 1 */
92 u8 countRateRecordTag; /* countrate tag = 1000 (4 bits) */
93 /* third record : Gate Digi record */
94 u8 gateDigiRecordBool; /* Gate single digi recorded if 1 */
95 u8 gateDigiRecordTag; /* countrate tag = 1100 (4 bits) */
96 /* + eventually other records */
97 };

139/* 1.G This structure gives what's in the gate digi record for building
140 the gate digi encoding pattern : TTTT CSpe rGRR RRRR
141TTTT = 1100
142
143C = 1 if number of compton stored
144p = 1 if Source decay XYZ pos stored
145S = 1 if source ID stored
146e = 1 if eventID stored
147r = 1 if runID stored
148G = 1 if global dig XYZ pos stored
149R = 0 reserved bit
150
151Time always stored
152*/
153
154 struct LMF_ccs_gateDigiHeader {
155
156 u8 comptonBool; /* C number of compton */
157 u8 comptonDetectorBool; /* D number of compton in detector */
158 u8 sourceIDBool; /* s */
159 u8 sourceXYZPosBool; /* S */
160 u8 eventIDBool; /* e */
161 u8 runIDBool; /* r */
162 u8 globalXYZPosBool; /* G global xyz pos */
163 u8 multipleIDBool; /* M multiple ID */
164 };
