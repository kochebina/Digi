387// Tenplate for finding moves of a specific type: the function returns a non-zero pointer
388// only if a move of the requested type was found in the creator's move list
389template <class C>
390C* GateSystemComponent::FindMove() const
391{
392 if (!m_creator)
393 return 0;
394 GateObjectRepeaterList* aList = m_creator->GetMoveList();
395 for (size_t i=0; i < aList->size() ; i++)
396 if ( dynamic_cast<C*>(aList->GetRepeater(i)) )
397 return dynamic_cast<C*>(aList->GetRepeater(i));
398 return 0;
399}
