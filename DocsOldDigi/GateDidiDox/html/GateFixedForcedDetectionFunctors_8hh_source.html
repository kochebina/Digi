203 void CreateMaterialMuMap(G4EmCalculator *emCalculator,
204 const std::vector<double> & energyList,
205 GateVImageVolume * gateImageVolume)
206 {
207 m_EnergyList = energyList;
208 itk::TimeProbe muProbe;
209 muProbe.Start();
210 /* Get image materials + world */
211 std::vector<G4Material*> imageWorldMaterials;
212 gateImageVolume->BuildLabelToG4MaterialVector(imageWorldMaterials);
213 GateVVolume *volume = gateImageVolume;
214 while (volume->GetLogicalVolumeName() != "world_log")
215 {
216 volume = volume->GetParentVolume();
217 }
218 imageWorldMaterials.push_back(const_cast<G4Material*>(volume->GetMaterial()));
219
220 /* Get the list of involved processes (Rayleigh, Compton, PhotoElectric) */
221 G4ParticleDefinition* particle = G4ParticleTable::GetParticleTable()->FindParticle("gamma");
222 G4ProcessVector* processList = particle->GetProcessManager()->GetProcessList();
223 std::vector<G4String> processNameVector;
224 for (unsigned int process = 0; process < processList->size(); process++)
225 {
226 G4ProcessType type = (*processList)[process]->GetProcessType();
227 std::string name = (*processList)[process]->GetProcessName();
228 if ((type == fElectromagnetic) && (name != "msc"))
229 {
230 processNameVector.push_back(name);
231 }
232 }
233
234 MaterialMuImageType::RegionType region;
235 region.SetSize(0, imageWorldMaterials.size());
236 region.SetSize(1, energyList.size());
237 m_MaterialMu = MaterialMuImageType::New();
238 m_MaterialMu->SetRegions(region);
239 m_MaterialMu->Allocate();
240 itk::ImageRegionIterator<MaterialMuImageType> it(m_MaterialMu, region);
241 for (unsigned int energy = 0; energy < energyList.size(); energy++)
242 {
243 for (unsigned int i = 0; i < imageWorldMaterials.size(); i++)
244 {
245 G4Material * mat = imageWorldMaterials[i];
246 double mu = 0;
247 for (unsigned int process = 0; process < processNameVector.size(); process++)
248 {
249 /* Note: the G4EmCalculator retrive the correct G4VProcess
250 (standard, Penelope, Livermore) from the processName. */
251 double crossSection = emCalculator->ComputeCrossSectionPerVolume(energyList[energy],
252 "gamma",
253 processNameVector[process],
254 mat->GetName());
255 /* In (length unit)^{-1} according to
256 http://www.lcsim.org/software/geant4/doxygen/html/classG4EmCalculator.html#a870d5fffaca35f6e2946da432034bd4c */
257 mu += crossSection;
258 }
259 it.Set(mu);
260 ++it;
261 }
262 }
263
264 muProbe.Stop();
265 G4cout
266 << "Computation of the mu lookup table took "
267 << muProbe.GetTotal() << ' '
268 << muProbe.GetUnit()
269 << G4endl;
270
271 MaterialMuImageType::SpacingType spacing;
272 spacing[0] = 1.;
273 if(energyList.size()>1)
274 spacing[1] = energyList[1]-energyList[0];
275 else
276 spacing[1] = 1.*keV;
277 m_MaterialMu->SetSpacing(spacing);
278 }

299 m_EnergyList = energyList;
300 itk::TimeProbe deltaProbe;
301 deltaProbe.Start();
302
303 /* Get image materials + world */
304 std::vector<G4Material*> imageWorldMaterials;
305 gateImageVolume->BuildLabelToG4MaterialVector(imageWorldMaterials);
306 GateVVolume *volume = gateImageVolume;
307 while (volume->GetLogicalVolumeName() != "world_log")
308 {
309 volume = volume->GetParentVolume();
310 }
311 imageWorldMaterials.push_back(const_cast<G4Material*>(volume->GetMaterial()));
312
313 MaterialDeltaImageType::RegionType region;
314 region.SetSize(0, imageWorldMaterials.size());
315 region.SetSize(1, energyList.size());

318 m_MaterialDelta->Allocate();

320 for (unsigned int energy = 0; energy < energyList.size(); energy++)

327#ifdef GATE_USE_XRAYLIB

333 if (error != NULL) {

335 xrl_clear_error(&error);
336 }
337 }
338#else
339 for (unsigned int i = 0; i < mat->GetElementVector()->size(); ++i)
340 delta += (1 - Refractive_Index_Re(mat->GetElementVector()->at(i)->GetSymbol(), energyList[energy]/(keV), 1.0)) * mat->GetFractionVector()[i];
341#endif
342#else
343 G4Exception( "GateFixedForcedDetectionFunctors::CreateMaterialDeltaMap", "CreateMaterialDeltaMap", FatalException, "Xraylib is not available\n");
344#endif
345 delta *= Density;
346 it.Set(delta);
347 ++it;
348 }
349 }
350
351 deltaProbe.Stop();
352 G4cout
353 << "Computation of the delta lookup table took "
354 << deltaProbe.GetTotal() << ' '
355 << deltaProbe.GetUnit()
356 << G4endl;

359 spacing[0] = 1.;
360 if(energyList.size()>1)
361 spacing[1] = energyList[1]-energyList[0];
362 else

539 inline void operator()(const rtk::ThreadIdType threadId,
540 const float &itkNotUsed(input),
541 float & output,
542 const double & itkNotUsed(rayCastValue),
543 const VectorType & stepInMM,
544 const VectorType & itkNotUsed(source),
545 const VectorType & sourceToPixel,
546 const VectorType & nearestPoint,
547 const VectorType & farthestPoint)
548 {
549 double *p = m_MaterialMu->GetPixelContainer()->GetBufferPointer();
550 /* Multiply interpolation weights by step norm in MM to convert voxel
551 intersection length to MM. */
552 const double stepInMMNorm = stepInMM.GetNorm();
553 for (unsigned int j = 0; j < m_InterpolationWeights[threadId].size() - 1; j++)
554 {
555 m_InterpolationWeights[threadId][j] *= stepInMMNorm;
556 }
557
558 /* The last material is the world material. One must fill the weight with
559 the length from source to nearest point and farthest point to pixel
560 point. */
561 VectorType worldVector = sourceToPixel + nearestPoint - farthestPoint;
562 for (int i = 0; i < 3; i++)
563 {
564 worldVector[i] *= m_VolumeSpacing[i];
565 }
566 m_InterpolationWeights[threadId].back() += worldVector.GetNorm();
567
568 /* Loops over energy, multiply weights by mu, accumulate using Beer Lambert */
569 for (unsigned int i = 0; i < m_EnergyWeightList->size(); i++)
570 {
571 double rayIntegral = 0.;
572 for (unsigned int j = 0; j < m_InterpolationWeights[threadId].size(); j++)
573 {
574 rayIntegral += m_InterpolationWeights[threadId][j] * *p++;
575 }
576
577 /* Statistical noise added */
578 if (m_NumberOfPrimaries != 0)
579 {
580 double a = std::exp(-rayIntegral);
581 double nprimE = m_NumberOfPrimaries * (*m_EnergyWeightList)[i];
582 double n = ((nprimE)?G4Poisson(nprimE*a)/nprimE:0.);
583 this->Accumulate(threadId,
584 output,
585 n * (*m_EnergyWeightList)[i] * (*m_ResponseDetector)(m_EnergyList[i] ),
586 m_EnergyList[i]);
587 }
588 else
589 {
590 this->Accumulate(threadId,
591 output,
592 std::exp(-rayIntegral) * (*m_EnergyWeightList)[i],
593 m_EnergyList[i]);
594 }
595
596 /* Phase for Fresnel diffraction */
597 if (m_MuToDeltaImageOffset != 0)
598 {
599 double *q = m_MaterialDelta->GetPixelContainer()->GetBufferPointer();
600 double rayIntegral = 0.;
601 for (unsigned int j = 0; j < m_InterpolationWeights[threadId].size(); j++)
602 {
603 rayIntegral += m_InterpolationWeights[threadId][j] * *q++;
604 }
605
606 /* Matter wave : for photon, the formula is lambda = hc/E
607 * https://fr.wikipedia.org/wiki/Hypoth%C3%A8se_de_De_Broglie
608 */
609 double wavelength = h_Planck/(eV*s) * c_light/(m/s) / (m_EnergyList[i]/eV);
610 wavelength = wavelength * (m/mm); // To use the same unit as Geant4
611 rayIntegral *= (-2*itk::Math::pi/wavelength);
612 this->AccumulatePhase(output,
613 rayIntegral + std::log((*m_EnergyWeightList)[i])/2.,
614 m_EnergyList[i]);
615 }
616 }
617
618 /* Reset weights for next ray in thread.*/
619 std::fill(m_InterpolationWeights[threadId].begin(),
620 m_InterpolationWeights[threadId].end(),
621 0.);
622 }

625 class ComptonValueAccumulation: public VAccumulation
626 {
627 public:
628
629 ComptonValueAccumulation()
630 {
631 /* G4 data */
632 G4VDataSetAlgorithm* scatterInterpolation = new G4LogLogInterpolation;
633 G4String scatterFile = "comp/ce-sf-";
634 m_ScatterFunctionData = new G4CompositeEMDataSet(scatterInterpolation, 1., 1.);
635 m_ScatterFunctionData->LoadData(scatterFile);
636
637 m_CrossSectionHandler = new G4CrossSectionHandler;
638 G4String crossSectionFile = "comp/ce-cs-";
639 m_CrossSectionHandler->LoadData(crossSectionFile);
640 }
641 ~ComptonValueAccumulation()
642 {
643 delete m_ScatterFunctionData;
644 delete m_CrossSectionHandler;
645 }
646
647 inline void operator()(const rtk::ThreadIdType threadId,
648 const float &itkNotUsed(input),
649 float &output,
650 const double &itkNotUsed(rayCastValue),
651 const VectorType &stepInMM,
652 const VectorType &itkNotUsed(source),
653 const VectorType &sourceToPixel,
654 const VectorType &nearestPoint,
655 const VectorType &farthestPoint)
656 {
657 /* Compute ray length in world material
658 This is used to compute the length in world as well as the direction
659 of the ray in mm. */
660 VectorType worldVector = sourceToPixel + nearestPoint - farthestPoint;
661 for (int i = 0; i < 3; i++)
662 {
663 worldVector[i] *= m_VolumeSpacing[i];
664 }
665 const double worldVectorNorm = worldVector.GetNorm();
666
667 /* This is taken from G4LivermoreComptonModel.cc */
668 double cosT = worldVector * m_Direction / worldVectorNorm;
669 double x = std::sqrt(1. - cosT) * m_InvWlPhoton; /* 1-cosT=2*sin(T/2)^2 */
670 double scatteringFunction = m_ScatterFunctionData->FindValue(x, m_Z - 1);
671
672 /* This is taken from GateDiffCrossSectionActor.cc and simplified */
673 double Eratio = 1. / (1. + m_E0m * (1. - cosT));
674 double DCSKleinNishina = m_eRadiusOverCrossSectionTerm
675 * Eratio
676 * (1. + Eratio * (Eratio - 1. + cosT * cosT));
677 double DCScompton = DCSKleinNishina * scatteringFunction;
678
679 /* Multiply interpolation weights by step norm in MM to convert voxel
680 intersection length to MM. */
681 const double stepInMMNorm = stepInMM.GetNorm();
682 for (unsigned int j = 0; j < m_InterpolationWeights[threadId].size() - 1; j++)
683 {
684 m_InterpolationWeights[threadId][j] *= stepInMMNorm;
685 }
686
687 /* The last material is the world material. One must fill the weight with
688 the length from farthest point to pixel point. */
689
690 if (!m_generatePhotons)
691 {
692 m_InterpolationWeights[threadId].back() = worldVectorNorm;
693 }
694 else
695 {
696 m_InterpolationWeights[threadId].back() = worldVectorNorm;
697 }
698 const double energy = Eratio * m_Energy;
699 unsigned int e = itk::Math::Round<double, double>(energy / m_MaterialMu->GetSpacing()[1]);
700 double *p = m_MaterialMu->GetPixelContainer()->GetBufferPointer()
701 + e * m_MaterialMu->GetLargestPossibleRegion().GetSize()[0];
702
703 /* Ray integral */
704 double rayIntegral = 0.;
705 for (unsigned int j = 0; j < m_InterpolationWeights[threadId].size(); j++)
706 {
707 rayIntegral += m_InterpolationWeights[threadId][j] * *p++;
708 }
709
710 /* Final computation */
711 // double weight = std::exp(-rayIntegral) * DCScompton * GetSolidAngle(sourceToPixel) * (*m_ResponseDetector)(energy);
712 double weight = std::exp(-rayIntegral) * DCScompton * GetSolidAngle(sourceToPixel);
713 if (m_generatePhotons)
714 {
715 //Accumulate(threadId, output, weight, m_Energy);
716 VectorType photonDirection;
717 VectorType photonPosition;
718 for (int i = 0; i < 3; i++)
719 {
720 photonDirection[i] = worldVector[i] / worldVectorNorm;
721 photonPosition[i] = farthestPoint[i] * m_VolumeSpacing[i];
722 }
723 SavePhotonsparameters(threadId, photonPosition, photonDirection, weight, energy);
724 }
725 else if (m_ARF)
726 {
727 VectorType photonDirection;
728 VectorType photonPosition;
729 for (int i = 0; i < 3; i++)
730 {
731 photonDirection[i] = worldVector[i] / worldVectorNorm;
