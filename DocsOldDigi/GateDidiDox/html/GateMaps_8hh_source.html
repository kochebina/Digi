109// Constructs a new GateMap by combining several GateMaps
110// Pass the number of GateMaps and an array of GateMap pointers
111template <typename Tkey,typename Tvalue>
112inline GateMap<Tkey,Tvalue>::GateMap(size_t n, GateMap<Tkey,Tvalue> *mapArray[])
113 : std::map<Tkey,Tvalue>()
114{
115 for (size_t i=0; i<n ; i++){
116 GateMap<Tkey,Tvalue>* mapElement = mapArray[i];
117 for (iterator iter = mapElement->begin(); iter != mapElement->end(); iter++)
118 this->insert(*iter);
119 }
120}

125// Dump a map into a printable string
126// This function concatenates all the map keys into one printable string
127// Each key can be preceded by a prefix and followed by a suffix. It can be printed within quotes.
128// Also, each key can be separated from the next one by a separator-string.
129template <typename Tkey,typename Tvalue>
130inline G4String GateMap<Tkey,Tvalue>::DumpMap(G4bool flagWithQuotes, G4String suffix, G4String separator, G4String prefix )
131{
132 G4String dump;
133
134 iterator iter = this->begin();
135 while (iter != this->end()) {
136 dump += prefix;
137 if (flagWithQuotes)
138 dump += "'";
139 dump += G4String(iter->first) + suffix;
140 if (flagWithQuotes)
141 dump += "'";
142 if ( (++iter) != this->end() )
143 dump += separator;
144 }
145 return dump;
146}
